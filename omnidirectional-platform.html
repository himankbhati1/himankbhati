Objective: To engineer a swarm-capable mobile robot featuring holonomic drive (movement in any direction) and a 4-DOF manipulator, controlled wirelessly via a custom Web Interface.

02. System Architecture

The robot operates on a distributed control architecture, splitting low-level motor driving from high-level logic.

Holonomic Drive

4-DOF Manipulator

Custom robotic arm designed for precise pick-and-place operations using Servo motors.

04. Embedded Firmware

The core logic runs on an ESP32 using the Arduino framework. It creates a SoftAP (Access Point) and serves a lightweight HTML control dashboard directly from memory.

Robotic_arm.ino 
#include <WiFi.h>
#include <WebServer.h>
#include <ESP32Servo.h>

const char* ssid = "ESP32_ARM";
const char* password = "12345678";

WebServer server(80);
Servo s1, s2, s3, s4;

// GPIO pins
int p1 = 13, p2 = 12, p3 = 14, p4 = 27;
// Initial Angles
int a1 = 90, a2 = 90, a3 = 90, a4 = 90;

// ---------------- HTML PAGE ----------------
String page = R"rawliteral(
<!DOCTYPE html>
<html>
<head>
<meta name="viewport" content="width=device-width, initial-scale=1">
<style>
body{ font-family: Arial; background:#0f172a; color:#e5e7eb; }
.card{ background:#1e293b; border-radius:10px; padding:10px; }
button{ background:#22c55e; border:none; border-radius:6px; }
</style>
</head>
<body>
<h2>ESP32 Robotic Arm</h2>
<!-- Slider Controls -->
<div class="card">
  <div class="title">Motor 1</div>
  <input id="s1" type="range" min="0" max="180" value="90" oninput="set(1,this.value)">
</div>
<script>
function set(m,val){
  fetch("/m"+m+"?v="+val);
}
</script>
</body>
</html>
)rawliteral";

void setup() {
  Serial.begin(115200);
  WiFi.softAP(ssid, password);
  
  // Define Routes
  server.on("/", [](){ server.send(200,"text/html",page); });
  
  server.on("/m1", [](){ 
    a1 = constrain(server.arg("v").toInt(),0,180); 
    s1.write(a1); 
    server.send(200); 
  });

  server.begin();
  
  // Attach Servos at 50Hz
  s1.setPeriodHertz(50);
  s1.attach(p1, 500, 2500);
}

void loop() {
  server.handleClient();
}
